\section{Programs}

\begin{frame}{Syntax}
  \begin{itemize}
  \item We will work with a simple imperative programming language.
  \item We start by defining how programs must be \emph{written}.
  \end{itemize}
  %
  \pause
  %
  \begin{block}{Syntax of expressions}
    \vspace*{-2em}
    \begin{flalign*}
      \texttt{x}, \texttt{y}, \ldots \quad \in \quad \mathbb{V} && \text{variables}\\
      \texttt{E} \quad \in \quad \mathbb{E} & \quad ::= \quad \texttt{1}~
                                              |~\texttt{x}~
                                              |~\texttt{E}_1 \texttt{ - } \texttt{E}_2
                                            & \text{arithmentic expressions}\\
                                            & \quad \quad| \;\;\:\quad \texttt{tt}~
                                              |~\texttt{ff}~
                                              |~\texttt{E}_1 \texttt{ < } \texttt{E}_2~
                                              |~\texttt{E}_1 \texttt{ nand } \texttt{E}_2
                                            & \text{Boolean expressions}
    \end{flalign*}
  \end{block}
  %
  \vspace{-1em}\pause
  %
  \begin{block}{Syntax of statements}
    \vspace*{-2em}
    \begin{align*}
        \texttt{S} \quad \in \quad \mathbb{S} & \quad
        ::= \quad \texttt{skip}~
          |~\texttt{x := E}~
          |~\texttt{S}_1 \texttt{; } \texttt{S}_2~
          |~\texttt{\textbf{if} (E) \textbf{then }} \texttt{S}_1 \texttt{\bfseries~else } \texttt{S}_2~
          |~\texttt{\textbf{while} (E) \textbf{do }} \texttt{S}
    \end{align*}
    \vspace*{-2em}
  \end{block}
  %
  \begin{itemize}
  \pause\item Purely \emph{syntactic} object, it has no \emph{meaning}
    (\emph{i.e.}, no semantics) yet.
  \pause\item Defined \emph{compositionally}, \emph{i.e.}, composite
    elements such as \texttt{E}\textsubscript{1}\texttt{ - }\texttt{E}\textsubscript{2}
    have a unique decomposition into \emph{strictly smaller} elements.
  \end{itemize}
\end{frame}
%
\begin{frame}{Semantics}
  \begin{itemize}
  \item We now turn our heads to the meaning of programs: \emph{semantics}.
  \item Semantics can come in many different flavors, and they complement each other.
  \end{itemize}
  %
  \pause\medskip
  %
  {\small\begin{minipage}[t]{.32\textwidth}
    \begin{center}
      \textbf{Operational\\\cite{plotkin-sopsem}}
    \end{center}
    %
    \vspace{-.8em}
    %
    \begin{itemize}
    \item Defines \emph{how} a program is executed.
    \item \emph{Small-} or \emph{big-step} definitions.
    \end{itemize}
  \end{minipage}
  %
  \hfill
  %
  \begin{minipage}[t]{.32\textwidth}
    \begin{center}
      \textbf{Denotational\\\cite{scott71}}
    \end{center}
    %
    \vspace{-.8em}
    %
    \begin{itemize}
    \item Models the \emph{result} of executing a program.
    \item \emph{Compositional} definitions.
    \end{itemize}
  \end{minipage}
  %
  \hfill
  %
  \begin{minipage}[t]{.32\textwidth}
    \begin{center}
      \textbf{Axiomatic\\\cite{hoare69}}
    \end{center}
    %
    \vspace{-.8em}
    %
    \begin{itemize}
    \item Allows to prove (partial) correctess of a program.
    \item Assertion-based definitions.
    \end{itemize}
  \end{minipage}}
  %
  \pause\bigskip
  %
  \begin{itemize}
  \item Herein, we concentrate in \emph{denotational} semantics, but
    the approaches can be adapted to any semantics flavor.
  \end{itemize}
  %
  % \item As \emph{numerals} (\emph{i.e.}, expressions) and
  %   \emph{numbers} (\emph{i.e.,} mathematical, abstract objects)
  %   differ, there is a distinction between the program itself and what
  %   it \emph{denotes}.
  % \end{itemize}
\end{frame}
%
\begin{frame}{Denotational semantics: expressions}
  \begin{itemize}
  \item We consider the \emph{structure}
    $(\mathbb{Z}, \mathbb{B}; \supInter{1}, \supInter{\mathfrak{tt}}, \supInter{\mathfrak{ff}},
    \supInter{-}, \supInter{<}, \supInter{\uparrow})$ with the usual interpretations.
  \item And, for brevity, let $\mathbb{I} \triangleq \mathbb{Z} \cup \mathbb{B}$, and
    $\mathbb{Ev} \triangleq \mathbb{V} \to \mathbb{I}$.
  \end{itemize}
  %
  \pause
  %
  \begin{block}{Denotational semantics of expressions}
    $\pmb{\mathscr{E}} : \mathbb{V} \to (\texttt{E} \in \mathbb{E})
    \to \mathbb{Ev} \rightharpoonup \mathbb{I}
    %
    \text{\small\itshape~where } \mathbb{vars}(\texttt{E}) \subseteq
    \mathbb{V}$\footnote{%
      We use a dependent type to express that we require the
      expression \texttt{E} to have all of its variables considered in the
      (implicit) set of variables $\mathbb{V}$.  A more detailed discussion on
      this matter can be found in~\cite{Cousot21-book}.%
    }
    %
    \begin{flalign*}
    \hspace*{-.5em}\begin{array}{lllll}
      \esem{\texttt{k}} \sigma
      &\triangleq
      &k
      &\in \mathbb{I}
      &\\[.5em]
      %
      \esem{\texttt{x}} \sigma
      &\triangleq
      &\sigma(\texttt{x})
      &\in \mathbb{I}
      &\\[.5em]
      %
      \esem{\texttt{E}_1 \texttt{ - } \texttt{E}_2} \sigma
      &\triangleq
      &v_1 \supInter{-} v_2
      &\in \mathbb{Z}
      &\text{%
        if $v_1 = \esem{\texttt{E}_1}\sigma \in \mathbb{Z}$,
           $v_2 = \esem{\texttt{E}_2}\sigma \in \mathbb{Z}$
      }\\[.5em]
      %
      \esem{\texttt{E}_1 \texttt{ < } \texttt{E}_2} \sigma
      &\triangleq
      &v_1 \supInter{<} v_2
      &\in \mathbb{B}
      &\text{%
        if $v_1 = \esem{\texttt{E}_1}\sigma \in \mathbb{Z}$,
           $v_2 = \esem{\texttt{E}_2}\sigma \in \mathbb{Z}$
      }\\[.5em]
      %
      \esem{\texttt{E}_1 \texttt{ nand } \texttt{E}_2} \sigma
      &\triangleq
      &v_1 \supInter{\uparrow} v_2
      &\in \mathbb{B}
      &\text{%
       if $v_1 = \esem{\texttt{E}_1}\sigma \in \mathbb{B}$,
           $v_2 = \esem{\texttt{E}_2}\sigma \in \mathbb{B}$
      }\\[.5em]
      %
      undefined &&&& otherwise
    \end{array}&&
    \end{flalign*}
  \end{block}
  \vspace*{-.6em}
\end{frame}
%
\begin{frame}{Denotational semantics: expressions (example)}
  The expression ``\texttt{(x-1) < (y-1) nand ff}'' under
  $\sigma=\{\texttt{x} \mapsto 3,\texttt{y} \mapsto 2\}$ evaluates to:
  %
  \begin{flalign*}
    \esem{\texttt{(x-1) < (y-1) nand ff}} \sigma
    \quad&=\quad \esem{\texttt{(x-1) < (y-1)}}\sigma \supInter{\uparrow} \esem{\texttt{ff}}\sigma\\
    \quad&=\quad \esem{\texttt{(x-1)}}\sigma \supInter{<} \esem{\texttt{(y-1)}}\sigma \supInter{\uparrow} \esem{\texttt{ff}}\sigma\\
    \quad&=\quad (\esem{\texttt{x}}\sigma \supInter{-} \esem{\texttt{1}}\sigma) \supInter{<} \esem{\texttt{y-1}}\sigma \supInter{\uparrow} \esem{\texttt{ff}}\sigma\\
    \quad&=\quad (\esem{\texttt{x}}\sigma \supInter{-} \esem{\texttt{1}}\sigma) \supInter{<} (\esem{\texttt{y}}\sigma \supInter{-} \esem{\texttt{1}}\sigma) \supInter{\uparrow} \esem{\texttt{ff}}\sigma\\
    \quad&=\quad (3 \supInter{-} \supInter{1}) \supInter{<} (2 \supInter{-} \supInter{1}) \supInter{\uparrow} \supInter{\mathfrak{ff}}\\
    \quad&=\quad 2 \supInter{<} \supInter{1} \supInter{\uparrow} \supInter{\mathfrak{ff}}\\
    \quad&=\quad \supInter{\mathfrak{ff}} \supInter{\uparrow} \supInter{\mathfrak{ff}}\\
    \quad&=\quad \supInter{\mathfrak{tt}}
  \end{flalign*}
  %
\end{frame}
%
\begin{frame}{Denotational semantics: statements}
  \begin{itemize}
  \item Now, we turn our heads to statements.
  \end{itemize}
  %
  \begin{block}{Denotational semantics of statements}
    $\pmb{\mathscr{S}} : \mathbb{V} \to (\texttt{S} \in \mathbb{S})
    \to \mathbb{Ev} \rightharpoonup \mathbb{Ev}
    %
    \text{\small\itshape~where } \mathbb{vars}(\texttt{S}) \subseteq
    \mathbb{V}$
    %
    \pause\begin{flalign*}
    \hspace*{-.5em}\begin{array}{llll}
      \ssem{\texttt{skip}} \sigma
      &\triangleq
      &\sigma
      &\\[.5em]
      %
      \ssem{\texttt{x := E}} \sigma
      &\triangleq
      &\sigma[v/\texttt{x}]
      &\text{%
        if $\esem{\texttt{E}}\sigma = v$
      }\\[.5em]
      %
      \ssem{\texttt{S}_1 \texttt{; } \texttt{S}_2} \sigma
      &\triangleq
      &(\ssem{\texttt{S}_2} \circ \ssem{\texttt{S}_1}) \sigma
      &\\[2em]
      %
      \ssem{\texttt{\textbf{if} (E) \textbf{then }} \texttt{S}_1 \texttt{\bfseries~else } \texttt{S}_2} \sigma
      &\triangleq
      &\begin{cases}
        \ssem{\texttt{S}_1} \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{tt}}\\
        \ssem{\texttt{S}_2} \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{ff}}\\
        undefined & otherwise
      \end{cases}
      &\\[.5em]
    \end{array}&&
    \end{flalign*}
  \end{block}
\end{frame}
%
\begin{frame}{Denotational semantics: loops}
  \begin{flalign*}
    \hspace*{-.5em}\begin{array}{lll}
      \ssem{\texttt{\textbf{while} (E) \textbf{do} S}} \sigma
      &\triangleq
      &\begin{cases}
        \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{ff}}\\
        (\ssem{\texttt{\only<2->{\color{red}}\textbf{while} (E) \textbf{do} S}} \circ \ssem{\texttt{S}}) \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{tt}}\\
        undefined & otherwise
      \end{cases}
    \end{array}&&
  \end{flalign*}
  %
  \begin{itemize}
  \pause\item \emph{\color{red}Recursive} definition!
  \pause\item We need to make sure that:
    \begin{enumerate}
    \item The equation above has (at least) one solution.
    \item If the equation has a solution, we must take the \emph{most precise} one.
    \end{enumerate}
  \pause\item And, for this, we use \emph{fixpoints} on \emph{posets} with \emph{information orders}.
  \end{itemize}
  %
\end{frame}
%
\begin{frame}{Posets \& fixpoints: recap}
  \vspace*{-1em}
  \begin{itemize}
  \item We start by defining a poset $(\mathcal{I},\leq_{\mathcal{I}})$ with:
    %
    \begin{align*}
    \mathcal{I} &\triangleq \mathbb{I}_\bot,\tag*{\scriptsize\color{gray} (values + undefined: $\mathbb{I} \cup \bot$)}\\
    x \leq_{\mathcal{I}} y &\triangleq x = \bot \vee x = y.\tag*{\scriptsize\color{gray} (information order on values)}
    \end{align*}
    %
    That is, we add a bottom-most element $\bot$, the \emph{undefined}
    value and we define a \emph{flat order}, yielding a nicely
    structured poset: directed-complete, and finite chains.
  \end{itemize}
  %
  \begin{center}
  \begin{tikzpicture}[node distance=.3cm]
    \node (bot) {$\bot$};
    \node (dots) [above=of bot] {$\ldots$};
    \node (two) [left=of dots] {$2$};
    \node (one) [left=of two] {$1$};
    \node (tt) [right=of dots] {$\supInter{\mathfrak{tt}}$};
    \node (ff) [right=of tt] {$\supInter{\mathfrak{ff}}$};
    \draw (bot) -- (one);
    \draw (bot) -- (two);
    \draw (bot) -- (tt);
    \draw (bot) -- (ff);
  \end{tikzpicture}
  \end{center}
  %
  \begin{itemize}
    \pause\item Similarly, we define the poset $(\mathcal{T},\leq_{\mathcal{T}})$
    with:
    %
    \begin{align*}
    \mathcal{T} &\triangleq \mathbb{Ev}_\bot \to \mathbb{Ev}_\bot,\tag*{\scriptsize\color{gray} (total, continuous state transformers)}\\
      f \leq_{\mathcal{T}} g &\triangleq \forall \sigma\:\pmb{:}\: f(\sigma) \; \dot{\leq}_{\mathcal{I}} \; g(\sigma).\tag*{\scriptsize\color{gray} (information order on state transformers)}
    \end{align*}
    %
    Where $\dot{\leq}_{\mathcal{I}}$ is the \emph{point-wise extension} of $\leq_{\mathcal{I}}$ to states, and
    $\pmb{\lambda}\sigma\pmb{.}\; \bot$ is the bottom-most element.
    %
    This poset preserves all the nice properties of the one above.
  \end{itemize}
\end{frame}
%
\begin{frame}{Posets \& fixpoints: recap (cont'd)}
  \vspace*{-1em}
  %
  \begin{itemize}
  \item In out context:
  \item the poset $(\mathcal{T},\leq_{\mathcal{T}})$ is a
    \emph{directed-complete partial order} (dcpo) since all of its
    directed subsets have a supremum.
    %
    That is,
    $\forall D \subseteq
    \mathcal{T},~directed\;\pmb{:}\;\vee_{\mathcal{T}}D \in
    \mathcal{T}$; and,
  \pause\item a function $F$ is \emph{Scott-continuous} in a poset
    $(\mathcal{T},\leq_{\mathcal{T}})$ if it preserves all directed
    suprema.
    %
    That is,
    $\forall D \subseteq \mathcal{T},~directed\;\pmb{:}\;F(\vee_{\mathcal{T}}D)
    = \vee_{\mathcal{T}}\{F(f)\;\pmb{|}\;f \in D\}.$
  \end{itemize}
  \pause\begin{theorem}[Kleene's theorem]
    Given a $\mathrm{directed}$-$\mathrm{complete~partial~order}$
    (dcpo) $(\mathcal{T},\leq_{\mathcal{T}})$, and a
    $\mathrm{Scott}$-$\mathrm{continuous~function}$
    $F : \mathcal{T} \to \mathcal{T}$, $f$ has a
    $\mathrm{least~fixpoint}$, which is the supremum:
    %
    \[
      \pmb{lfp}_{\leq_{\mathcal{T}}} F = \vee_{\mathcal{T}} \{F^n(\pmb{\lambda}\sigma\pmb{.}\;\bot)\;\pmb{|}\;n < \omega\}
    \]
    %
    of the ascending
    Kleene chain:
    %
    \[
      F^0(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}}
      F^1(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}}
      F^2(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}}
      \ldots \leq_{\mathcal{T}}
      F^n(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}}
      \ldots
    \]
  \end{theorem}
  %
  \pause So, our goal now is to make our denotational semantics function
  $\pmb{\mathscr{S}}$ to ``fit'' in this setting.
\end{frame}
%
\begin{frame}{Loop fixpoint semantics}
  \begin{enumerate}
  \item Extend the definitions of $\pmb{\mathscr{E}}$ and $\pmb{\mathscr{S}}$ with
    $\bot$ in the co-domain to model ``undefined'' behavior and turn them into \emph{total functions}:
    %
    \[
      \pmb{\mathscr{S}} : \mathbb{V} \to (\texttt{S} \in \mathbb{S})
      \to \mathbb{Ev}\;{\color{MyPurple}\to}\;
      {\color{MyPurple}\mathbb{Ev}_\bot}
      %
      \text{\small\it~where } \mathbb{vars}(\texttt{S}) \subseteq
      \mathbb{V}.
    \]
  \pause\item Make them composable (\emph{i.e.}, domain $=$ co-domain) by allowing
    $\bot$ in the domain:
    %
    \[
      \pmb{\mathscr{S}} : \mathbb{V} \to (\texttt{S} \in \mathbb{S})
      \to {\color{MyPurple}\mathbb{Ev}_\bot} \to
      \mathbb{Ev}_\bot
      %
      \text{\small\it~where } \mathbb{vars}(\texttt{S}) \subseteq
      \mathbb{V}.
    \]
    \pause\item Prove that
    $\ssem{\texttt{\_}} : \mathbb{Ev}_\bot \to \mathbb{Ev}_\bot$
    is \emph{Scott-continuous}.\\
    {\small\color{gray}(Sketch: by structural induction on the
      program's syntax)}
  \end{enumerate}
\end{frame}
%
\begin{frame}{Loop fixpoint semantics (cont'd)}
  %
  Now, we can express the solution to the \texttt{\bfseries while}'s (semantic)
  equation as the \emph{least fixpoint} (\emph{i.e.}, most-precise for
  the information order) of the \emph{functional} $F$:
  %
  \begin{flalign*}
    \hspace*{-.5em}\begin{array}{lll}
      \ssem{\texttt{\textbf{while} (E) \textbf{do} S}} \sigma
      &\triangleq
      &\pmb{lfp}_{\leq_{\mathcal{T}}} F
    \end{array}&&
  \end{flalign*}
  %
  \vspace*{-2em}
  %
  \begin{flalign*}
    \hspace*{-.5em}\begin{array}{lll}
      \textit{where} &F : (\mathbb{Ev}_\bot \to \mathbb{Ev}_\bot) \to (\mathbb{Ev}_\bot \to \mathbb{Ev}_\bot),
      \hspace*{15em}\text{\small\color{gray} (or, $F : \mathcal{T} \to \mathcal{T}$)}\\[1em]
      &F(f)(\sigma) = \begin{cases}
        \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{ff}}\\
        (f \circ \ssem{\texttt{S}}) \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{tt}}\\
        \bot & otherwise
      \end{cases}
    \end{array}&&
  \end{flalign*}
\end{frame}
%
\begin{frame}[fragile]{Loop fixpoint semantics: example}
  \begin{itemize}
  \item Consider computing the semantics of the program:
\begin{lstlisting}[basicstyle=\ttfamily,columns=fullflexible,escapeinside={(*}{*)}]
  (*\textbf{while}*) (0 < n) (*\textbf{do}*)
    x := x - n; n := n - 1
\end{lstlisting}
  \item We know that $\ssem{\texttt{x := x - n; n := n - 1}}\sigma = \sigma[x-n/x,n-1/n]$.
  \item We compute the least fixpoint of the functional $F$ defined as follows:
    %
    \[
      F(f)(\sigma) = \begin{cases}
        \sigma & \text{if } \esem{\texttt{0 < n}} \sigma = \supInter{\mathfrak{ff}}\\
        f \circ \sigma[x-n/x,n-1/n] & \text{if } \esem{\texttt{0 < n}} \sigma = \supInter{\mathfrak{tt}}\\
        \bot & otherwise
      \end{cases}
    \]
    %
    by building the Kleene ascending chain $F^0(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}} F^1(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}} F^2(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}} \ldots$
  \end{itemize}
\end{frame}

\section{Properties}
\begin{frame}{Properties as sets}
  \begin{itemize}
  \item Given a set of entities (or universe) $\mathscr{U}$,
    properties are \emph{sets of entities} in $\mathscr{U}$ belonging
    to $\wp(\mathscr{U})$ that have these properties.
  %
  \pause\item An entity $e \in \mathscr{U}$ has a property
    $P \in \wp(\mathscr{U})$ iff $e \in P$, or equivalently,
    $\{e\} \subseteq P$ (implication).
  %
  \pause\item $\text{\O}$ is the \emph{strongest} property.  \hfill {\small\color{gray}(false)}
  %
  % \item $\{e\}$ is the \emph{strongest} property of an entity.  \hfill
  %   {\small\color{gray}($e$ and nothing else)}
  %
  \item $\mathscr{U}$ is the \emph{weakest} property.  \hfill {\small\color{gray}(true)}
  %
  \item $\subseteq$ is \emph{implication}.  \hfill {\small\color{gray}($\to$)}
  \end{itemize}
  %
  \pause Two important remarks:
  %
  \begin{itemize}
  \item Herein, $e \in P \Leftrightarrow \{e\} \subseteq P$, and,
    although it may seem weird at first, the latter is the preferred
    one, since: (\textbf{i}) the use of $\subseteq$ allows us to move
    from set theory to order/lattice theory (monotone functions, fixpoints, \dots), and (\textbf{ii})
    abstracting $\subseteq$ is way easier than $\in$.
  \pause\item By considering properties as sets, we give raise
    to a Boolean algebra, since we are dealing with powerset-shaped
    (complete) lattices.
  \end{itemize}
  %
\end{frame}
%
\begin{frame}{Properties as sets: example}
  For example, take our universe $\mathscr{U}$ to be $\mathbb{Z}$, so
  that properties are in $\wp(\mathbb{Z})$.
  %
  \begin{itemize}
  \item $\text{\O}$ is false (no integer has this property).
  \item $\mathbb{Z}$ is true (all integers have this property).
  \item
    $2\mathbb{Z} \triangleq \{z \in \mathbb{Z}\;\pmb{|}\;\exists k \in
    \mathbb{Z}\,\pmb{:}\,z = 2k\}$ is the even property.
  \item
    $2\mathbb{Z}+1 \triangleq \{z \in \mathbb{Z}\;\pmb{|}\;\exists k
    \in \mathbb{Z}\,\pmb{:}\,z = 2k+1\}$ is the odd property.
  \end{itemize}
  %
  Intuitively, and integer can only be even or odd, expressed as:
  $\mathbb{Z} = 2\mathbb{Z} \cup 2\mathbb{Z}+1$.
\end{frame}

\section{Abstract Interpretation}
\begin{frame}{Motivation: undecidability}
  \begin{center}
    Proving non-trivial, program semantic properties is
    \textsc{undecidable} in general.
  \end{center}
  %
  \pause\bigskip
  %
  \emph{Abstract interpretation} formalizes a theory for computing
  sound approximations of program semantic properties.
  %
  In general, we (roughly) follow the following steps:
  %
  %

  %
  \bigskip
  %
  \pause{\color{gray} Abstract domains}
  \begin{itemize}
  \item Consider a subset of the \emph{concrete domain} (in our case,
    $(\wp(\mathbb{Ev}_\bot),\subseteq)$ of \emph{properties of
      interest}, and encode such properties in an \emph{abstract
      domain}
    $(\pmb{\mathfrak{A}},\preccurlyeq)$.
  \end{itemize}
  %
  \pause{\color{gray} Galois connections \& fixpoint abstraction}
  \begin{itemize}
  \item Establish a connection between concrete properties
    $P \in \wp(\mathbb{Ev}_\bot)$ and abstract properties
    $P^\sharp \in \mathfrak{A}$ such that proofs in the abstract
    \emph{imply} proofs in the concrete.
  \end{itemize}
  %
  \pause{\color{gray} Transformer abstraction}
  \begin{itemize}
  \item Design an abstract semantic transformer $\asem{\texttt{P}}$
    that over-approximates the concrete (collecting) semantic
    transformer $\csem{\texttt{P}}$.
  \end{itemize}
  % \item Establish a connection between 
  % \item This is where the technique of \emph{abstract interpretation} comes in!
\end{frame}
%
\begin{frame}[fragile]{Abstract domains: interval abstraction}
  \begin{itemize}
  \item We'll design an abstract domain for \emph{interval analysis}.
  \item Useful for, \emph{e.g.}, detecting out-of-bound array
    accesses.
  \end{itemize}
  %
\begin{lstlisting}[basicstyle=\ttfamily,columns=fullflexible,escapeinside={(*}{*)}]
  n := 1000;
  (*\color{MyDarkBlue}$\{\texttt{n} \in [1000,1000]\}$*)
  (*\textbf{while}*) (0 < n) (*\textbf{do}*)
    (*\color{MyDarkBlue}$\{\texttt{n} \in [1,1000]\}$*)
    n := n - 1
    (*\color{MyDarkBlue}$\{\texttt{n} \in [0,999]\}$*)
  (*\color{MyDarkBlue}$\{\texttt{n} \in [0,0]\}$*)
\end{lstlisting}
\end{frame}
%
\begin{frame}{Interval abstraction: the lattice}
  \begin{itemize}
  \item The set $\pmb{\mathfrak{I}}$ is defined as:
    %
    \(
      \pmb{\mathfrak{I}} \triangleq \{\bot\} \cup
      \{[\ell,r]\;\pmb{|}\;\ell \in \mathbb{Z}^{-\infty}, r \in
      \mathbb{Z}^{+\infty},\ell \leq r\}.
    \)
  \pause\item its order relation
    $\preccurlyeq$ as:
    %
    \(
      [\ell_1,r_1] \preccurlyeq [\ell_2,r_2] \Leftrightarrow \ell_2 \leq \ell_1 \wedge r_1 \leq r_2.
    \)
  \pause\item its \emph{lub} $\curlyvee$ as:
    %
    \(
      [\ell_1,r_1] \curlyvee [\ell_2,r_2] \triangleq [min\{\ell_1,\ell_2\},max\{r_1,r_2\}]
    \), with $\bot$ being the neutral element.
  \pause\item and its \emph{glb} $\curlywedge$ as:
    %
    \(
      [\ell_1,r_1] \curlywedge [\ell_2,r_2] \triangleq \begin{cases}
        \bot & \text{\textit{if} } r_1 < \ell_2 \vee r_2 < \ell_1\\
        [max\{\ell_1,\ell_2\},min\{r_1,r_2\}] & \text{\emph{otherwise}}
      \end{cases}
    \), with $\bot$ being the absorbing element.
  \pause\item Under this setting,
    $(\pmb{\mathfrak{I}},\preccurlyeq)$ is a \emph{complete}
    lattice.
  \pause\item And we point-wise extend it to
    $(\mathbb{Ev}^\sharp,\dot{\preccurlyeq})$ where
    $\mathbb{Ev}^\sharp \triangleq \mathbb{V} \to
    \pmb{\mathfrak{I}}$, and the relations/operators are adapted
    acoordignly, while preserving a complete lattice.
  \end{itemize}
\end{frame}
%
\begin{frame}{Interval abstraction: minus operation}
  \begin{itemize}
  \item The interval minus operation
    $-^\sharp : \pmb{\mathfrak{I}} \times \pmb{\mathfrak{I}} \to
    \pmb{\mathfrak{I}}$ defines the interval
    $[\ell_1,r_1] -^\sharp [\ell_2,r_2]$ of \texttt{x~-~y} when
    \texttt{x} has the interval $[\ell_1,r_1]$, and \texttt{y} has the
    interval $[\ell_2,r_2]$.
  \pause\item It is defined as follows:
    %
    \[
      [\ell_1,r_1] -^\sharp [\ell_2,r_2] \triangleq [\ell_1 - r_2, r_1 - \ell_2]
    \]
  \end{itemize}
\end{frame}
%
\begin{frame}{Interval abstraction: abstract semantics of expressions}
  \vspace*{-1em}
  % \begin{itemize}
  % \item The interval of an expression depends on the intervals of its variables.
  % \item We define the \emph{strict} abstract semantics of expressions.
  % \end{itemize}
  %
  \begin{block}{Abstract semantics of expressions}
    $\pmb{\mathscr{E}}^\sharp : \mathbb{V} \to (\texttt{E} \in \mathbb{E})
    \to \mathbb{Ev}^\sharp \to \pmb{\mathfrak{I}}
    %
    \text{\small\itshape~where } \mathbb{vars}(\texttt{E}) \subseteq
    \mathbb{V}$
    %
    \begin{flalign*}
    \hspace*{-.5em}\begin{array}{lllll}
      \aesem{\texttt{E}} \sigma^\sharp
      &\triangleq
      &\bot
      &\in \pmb{\mathfrak{I}}
      \quad\text{if some variable in $\sigma^\sharp$ has the interval $\bot$}
      &\\[.5em]
      %
      \aesem{\texttt{1}} \sigma^\sharp
      &\triangleq
      &[1,1]
      &\in \pmb{\mathfrak{I}}
      &\\[.5em]
      %
      \aesem{\texttt{tt}} \sigma^\sharp
      &\triangleq
      &[1,1]
      &\in \pmb{\mathfrak{I}}
      &\\[.5em]
      %
      \aesem{\texttt{ff}} \sigma^\sharp
      &\triangleq
      &[0,0]
      &\in \pmb{\mathfrak{I}}
      &\\[.5em]
      %
      \aesem{\texttt{x}} \sigma^\sharp
      &\triangleq
      &\sigma^\sharp(\texttt{x})
      &\in \pmb{\mathfrak{I}}
      &\\[.5em]
      %
      \aesem{\texttt{E}_1 \texttt{ - } \texttt{E}_2} \sigma^\sharp
      &\triangleq
      &\aesem{\texttt{E}_1}\sigma^\sharp -^\sharp \aesem{\texttt{E}_2}\sigma^\sharp
      &\in \pmb{\mathfrak{I}}
      &\\[.5em]
      %
      \aesem{\texttt{E}_1 \texttt{ < } \texttt{E}_2} \sigma^\sharp
      &\triangleq
      &\aesem{\texttt{E}_1}\sigma^\sharp \prec \aesem{\texttt{E}_2}\sigma^\sharp
      &\in \{[0,0],[1,1],[0,1]\}
      &\\[.5em]
      \bot &&& otherwise&
    \end{array}&&
    \end{flalign*}
  \end{block}
\end{frame}

\end{frame}

    \asem{\texttt{\textbf{while} (E) \textbf{do} S}} \sigma^\sharp
    \quad\triangleq\quad
    \pmb{lfp}_{\dot{\preccurlyeq}}\,F^\sharp \text{ \it with }\\[1em]

    \qquad F^\sharp(f^\sharp)(\sigma^\sharp) = \sigma^\sharp \curlyvee \begin{cases}
      (f^\sharp \circ \asem{\texttt{S}}) \sigma^\sharp & \text{if } 1 \in \aesem{E}\sigma^\sharp\\

    \end{cases}
  \end{array}&&
  \end{flalign*}
\end{frame}

\begin{frame}{Interval abstraction: soundness}
  \begin{itemize}
  \item We would like to prove that, for a given set of initial states
    $\Sigma$, the \emph{interval} semantics $\asem{\texttt{P}}$ is a
    \emph{weaker property} than the \emph{collecting} semantics
    $\csem{\texttt{P}}$.
  %
  \pause\item For that, we need to map abstract properties in
    $\mathbb{Ev}^\sharp$ to concrete properties in
    $\wp(\mathbb{Ev}_\bot)$, via the \emph{concretization} function
    %
    \[
      \dot{\gamma} : \mathbb{Ev}^\sharp \to \wp(\mathbb{Ev}_\bot),
    \]
    %
    and vice-versa, via the \emph{abstraction} function
    %
    \[
      \dot{\alpha} : \wp(\mathbb{Ev}_\bot) \to \mathbb{Ev}^\sharp.
    \]
  %
  \pause\item And they should yield \emph{equivalent} counterparts.
  \pause\item Formally,
    %
    \[
      \csem{\texttt{P}} \Sigma \subseteq (\dot{\gamma} \circ
      \asem{\texttt{P}} \circ \dot{\alpha}) \Sigma.
    \]
  \end{itemize}
\end{frame}
%
\begin{frame}{Interval abstraction: concretization}
  We define $\dot{\gamma}$ in steps.
  %
  First, $\gamma : \pmb{\mathfrak{I}} \to \wp(\mathbb{I}_\bot)$, where:
  %
  \begin{align*}
    \gamma(\bot) &\triangleq \text{\O}\\
    \gamma([\ell,r]) &\triangleq \{z \in \mathbb{Z}\;\pmb{|}\;\ell \leq z \leq r\}
  \end{align*}
  %
  Then, we point-wise extend $\dot{\gamma} : \mathbb{Ev}^\sharp \to \wp(\mathbb{Ev}_\bot)$, where:
  %
  \begin{align*}
    \dot{\gamma}(\sigma^\sharp) &\triangleq \{\sigma \in \mathbb{Ev}_\bot\;\pmb{|}\;\forall \texttt{x} \in \mathbb{V}\;\pmb{:}\;\sigma(\texttt{x}) \in \gamma(\sigma^\sharp(\texttt{x}))\}
  \end{align*}
\end{frame}
%
\begin{frame}{Interval abstraction: abstraction}
  We define $\dot{\alpha}$ in steps.
  %
  First, $\alpha: \wp(\mathbb{I}_\bot) \to \pmb{\mathfrak{I}}$, where:
  %
  \begin{align*}
    \alpha(\text{\O}) &\triangleq \bot\\
    \alpha(Z) &\triangleq [min\,Z,max\,Z]
  \end{align*}
  %
  Then, we point-wise extend $\dot{\alpha} : \wp(\mathbb{Ev}_\bot) \to \mathbb{Ev}^\sharp$, where:
  %
  \begin{align*}
    \dot{\alpha}(\Sigma) &\triangleq \{
                           \texttt{x} \mapsto [
                           min\{\sigma(\texttt{x})\;\pmb{|}\;\sigma \in \Sigma\},
                           max\{\sigma(\texttt{x})\;\pmb{|}\;\sigma \in \Sigma\}
                           ]\;\pmb{|}\; \texttt{x} \in \mathbb{V} \}
  \end{align*}
\end{frame}
%
\begin{frame}{Interval abstraction: Galois connection}
  \begin{itemize}
  \item The abstraction/concretization functions
    $(\dot{\alpha},\dot{\gamma})$ are closely related in that, for every
    $P \in \wp(\mathbb{Ev}_\bot)$ and $P^\sharp \in \mathbb{Ev}^\sharp$,
    %
    \[
      \dot{\alpha}(P)\:\dot{\preccurlyeq}\: P^\sharp \Leftrightarrow P
      \subseteq \dot{\gamma}(P^\sharp).
    \]
  \pause\item Which is the definition of a Galois connection (GC), which we write:
    %
    \[
      (\wp(\mathbb{Ev}_\bot),\subseteq)
      \galois{\dot{\alpha}}{\dot{\gamma}}
      (\mathbb{Ev}^\sharp,\dot{\preccurlyeq})
    \]
  \pause\item Intuitively, the implication $\dot{\alpha}(P)\:\dot{\preccurlyeq}\:P^\sharp \Rightarrow P
    \subseteq \dot{\gamma}(P^\sharp)$ states that $\dot{\alpha}(P)$ is a sound over-approximation
    of the property $P$; and,
  %
  \pause\item the converse implication
    $\dot{\alpha}(P)\:\dot{\preccurlyeq}\:P^\sharp \Leftarrow P
    \subseteq \dot{\gamma}(P^\sharp)$ states that $\dot{\alpha}(P)$ is the
    \emph{best} sound overapproximation of the property $P$.
  %
  \pause\item So, GCs model the over-approximation of program semantics.
  \end{itemize}
\end{frame}
%
\begin{frame}[fragile]{Interval abstraction: example}
  \vspace*{-1em}
  Going back to our example program \texttt{P}:
  %
\begin{lstlisting}[basicstyle=\ttfamily,columns=fullflexible,escapeinside={(*}{*)}]
  n := 1000;
  (*\textbf{while}*) (0 < n) (*\textbf{do}*)
    n := n - 1
\end{lstlisting}
  %
  with $\sigma^\sharp = \{\texttt{x} \mapsto [-\infty,+\infty]\}$, the
  abstract semantics would look like:
  %
  \begin{flalign*}
    \asem{\texttt{P}} \sigma^\sharp
    \quad&=\quad (\asem{\texttt{\textbf{while} (0<n) \textbf{do} n := n-1}}\sigma^\sharp \circ \asem{\texttt{n := 1000}}) \sigma^\sharp\\
    \quad&=\quad \asem{\texttt{\textbf{while} (0<n) \textbf{do} n := n-1}}\{\texttt{x} \mapsto [1000,1000]\}\\
    \quad&\ldots\quad \{\texttt{x} \mapsto ([1000,1000] - [1,1])\}\\
    \quad&\ldots\quad \{\texttt{x} \mapsto ([999,999] - [1,1])\}\\
    % \quad&\ldots\quad \{\texttt{x} \mapsto ([998,998] - [1,1])\}\\
    \quad&\ldots\quad \text{\itshape fixpoint convergence (using widening: pushing the interval ends}\\
    \quad&\qquad\; \text{\itshape to the constants in the program)}\\
    \quad&=\quad \{\texttt{x} \mapsto [0,1000]\}
  \end{flalign*}
  %
  Then,
  $\dot{\gamma}(\{\texttt{x} \mapsto [0,1000]\}) = \{\{\texttt{x}
  \mapsto 0\}, \{\texttt{x} \mapsto 1\}, \ldots\}$, which is weaker than
  $\csem{\texttt{P}}\{\texttt{x} \mapsto \mathbb{Z}\}$.
\end{frame}
%

  \item We can improve it by introducing context-sensitive semantics that takes into account
    the conditions when branching.
  \end{itemize}
