\section{Programs}

\begin{frame}{Syntax}
  \begin{itemize}
  \item We will work with a simple imperative programming language.
  \item We start by defining how programs must be \emph{written}.
  \end{itemize}
  %
  \pause
  %
  \begin{block}{Syntax of expressions}
    \vspace*{-2em}
    \begin{flalign*}
      \texttt{x}, \texttt{y}, \ldots \quad \in \quad \mathbb{V} && \text{variables}\\
      \texttt{E} \quad \in \quad \mathbb{E} & \quad ::= \quad \texttt{1}~
                                              |~\texttt{x}~
                                              |~\texttt{E}_1 \texttt{ - } \texttt{E}_2
                                            & \text{arithmentic expressions}\\
                                            & \quad \quad| \;\;\:\quad \texttt{tt}~
                                              |~\texttt{ff}~
                                              |~\texttt{E}_1 \texttt{ < } \texttt{E}_2~
                                              |~\texttt{E}_1 \texttt{ nand } \texttt{E}_2
                                            & \text{Boolean expressions}
    \end{flalign*}
  \end{block}
  %
  \vspace{-1em}\pause
  %
  \begin{block}{Syntax of statements}
    \vspace*{-2em}
    \begin{align*}
        \texttt{S} \quad \in \quad \mathbb{S} & \quad
        ::= \quad \texttt{skip}~
          |~\texttt{x := E}~
          |~\texttt{S}_1 \texttt{; } \texttt{S}_2~
          |~\texttt{\textbf{if} (E) \textbf{then }} \texttt{S}_1 \texttt{\bfseries~else } \texttt{S}_2~
          |~\texttt{\textbf{while} (E) \textbf{do }} \texttt{S}
    \end{align*}
    \vspace*{-2em}
  \end{block}
  %
  \begin{itemize}
  \pause\item Purely \emph{syntactic} object, it has no \emph{meaning}
    (\emph{i.e.}, no semantics) yet.
  \pause\item Defined \emph{compositionally}, \emph{i.e.}, composite
    elements such as \texttt{E}\textsubscript{1}\texttt{ - }\texttt{E}\textsubscript{2}
    have a unique decomposition into \emph{strictly smaller} elements.
  \end{itemize}
\end{frame}
%
\begin{frame}{Semantics}
  \begin{itemize}
  \item We now turn our heads to the meaning of programs: \emph{semantics}.
  \item Semantics can come in many different flavors, and they complement each other.
  \end{itemize}
  %
  \pause\medskip
  %
  {\small\begin{minipage}[t]{.32\textwidth}
    \begin{center}
      \textbf{Operational\\\cite{plotkin-sopsem}}
    \end{center}
    %
    \vspace{-.8em}
    %
    \begin{itemize}
    \item Defines \emph{how} a program is executed.
    \item \emph{Small-} or \emph{big-step} definitions.
    \end{itemize}
  \end{minipage}
  %
  \hfill
  %
  \begin{minipage}[t]{.32\textwidth}
    \begin{center}
      \textbf{Denotational\\\cite{scott71}}
    \end{center}
    %
    \vspace{-.8em}
    %
    \begin{itemize}
    \item Models the \emph{result} of executing a program.
    \item \emph{Compositional} definitions.
    \end{itemize}
  \end{minipage}
  %
  \hfill
  %
  \begin{minipage}[t]{.32\textwidth}
    \begin{center}
      \textbf{Axiomatic\\\cite{hoare69}}
    \end{center}
    %
    \vspace{-.8em}
    %
    \begin{itemize}
    \item Allows to prove (partial) correctess of a program.
    \item Assertion-based definitions.
    \end{itemize}
  \end{minipage}}
  %
  \pause\bigskip
  %
  \begin{itemize}
  \item Herein, we concentrate in \emph{denotational} semantics, but
    the approaches can be adapted to any semantics flavor.
  \end{itemize}
  %
  % \item As \emph{numerals} (\emph{i.e.}, expressions) and
  %   \emph{numbers} (\emph{i.e.,} mathematical, abstract objects)
  %   differ, there is a distinction between the program itself and what
  %   it \emph{denotes}.
  % \end{itemize}
\end{frame}
%
\begin{frame}{Denotational semantics: expressions}
  \begin{itemize}
  \item We consider the \emph{structure}
    $(\mathbb{Z}, \mathbb{B}; \supInter{1}, \supInter{\mathfrak{tt}}, \supInter{\mathfrak{ff}},
    \supInter{-}, \supInter{<}, \supInter{\uparrow})$ with the usual interpretations.
  \item And, for brevity, let $\mathbb{I} \triangleq \mathbb{Z} \cup \mathbb{B}$, and
    $\mathbb{Env} \triangleq \mathbb{V} \to \mathbb{I}$.
  \end{itemize}
  %
  \begin{block}{Denotational semantics of expressions}
    $\pmb{\mathscr{E}} : \mathbb{V} \to (\texttt{E} \in \mathbb{E})
    \to \mathbb{Env} \rightharpoonup \mathbb{I}
    %
    \text{\small\itshape~where } \mathbb{vars}(\texttt{E}) \subseteq
    \mathbb{V}$\footnote{%
      We use a dependent type to express that we require the
      expression \texttt{E} to have all of its variables considered in the
      (implicit) set of variables $\mathbb{V}$.  A more detailed discussion on
      this matter can be found in~\cite{Cousot21-book}.%
    }
    %
    \pause\begin{flalign*}
    \hspace*{-.5em}\begin{array}{lllll}
      \esem{\texttt{k}} \sigma
      &\triangleq
      &k
      &\in \mathbb{I}
      &\\[.5em]
      %
      \esem{\texttt{x}} \sigma
      &\triangleq
      &\sigma(\texttt{x})
      &\in \mathbb{I}
      &\\[.5em]
      %
      \esem{\texttt{E}_1 \texttt{ - } \texttt{E}_2} \sigma
      &\triangleq
      &v_1 \supInter{-} v_2
      &\in \mathbb{Z}
      &\text{%
        if $v_1 = \esem{\texttt{E}_1}\sigma \in \mathbb{Z}$,
           $v_2 = \esem{\texttt{E}_2}\sigma \in \mathbb{Z}$
      }\\[.5em]
      %
      \esem{\texttt{E}_1 \texttt{ < } \texttt{E}_2} \sigma
      &\triangleq
      &v_1 \supInter{<} v_2
      &\in \mathbb{B}
      &\text{%
        if $v_1 = \esem{\texttt{E}_1}\sigma \in \mathbb{Z}$,
           $v_2 = \esem{\texttt{E}_2}\sigma \in \mathbb{Z}$
      }\\[.5em]
      %
      \esem{\texttt{E}_1 \texttt{ nand } \texttt{E}_2} \sigma
      &\triangleq
      &v_1 \supInter{\uparrow} v_2
      &\in \mathbb{B}
      &\text{%
       if $v_1 = \esem{\texttt{E}_1}\sigma \in \mathbb{B}$,
           $v_2 = \esem{\texttt{E}_2}\sigma \in \mathbb{B}$
      }\\[.5em]
      %
      undefined &&&& otherwise
    \end{array}&&
    \end{flalign*}
  \end{block}
  \vspace*{-.6em}
\end{frame}
%
\begin{frame}{Denotational semantics: expressions (example)}
  The expression ``\texttt{(x-1) < (y-1) nand ff}'' under
  $\sigma=\{\texttt{x} \mapsto 3,\texttt{y} \mapsto 2\}$ evaluates to:
  %
  \begin{flalign*}
    \esem{\texttt{(x-1) < (y-1) nand ff}} \sigma
    \quad&=\quad \esem{\texttt{(x-1) < (y-1)}}\sigma \supInter{\uparrow} \esem{\texttt{ff}}\sigma\\
    \quad&=\quad \esem{\texttt{(x-1)}}\sigma \supInter{<} \esem{\texttt{(y-1)}}\sigma \supInter{\uparrow} \esem{\texttt{ff}}\sigma\\
    \quad&=\quad (\esem{\texttt{x}}\sigma \supInter{-} \esem{\texttt{1}}\sigma) \supInter{<} \esem{\texttt{y-1}}\sigma \supInter{\uparrow} \esem{\texttt{ff}}\sigma\\
    \quad&=\quad (\esem{\texttt{x}}\sigma \supInter{-} \esem{\texttt{1}}\sigma) \supInter{<} (\esem{\texttt{y}}\sigma \supInter{-} \esem{\texttt{1}}\sigma) \supInter{\uparrow} \esem{\texttt{ff}}\sigma\\
    \quad&=\quad (3 \supInter{-} \supInter{1}) \supInter{<} (2 \supInter{-} \supInter{1}) \supInter{\uparrow} \supInter{\mathfrak{ff}}\\
    \quad&=\quad 2 \supInter{<} \supInter{1} \supInter{\uparrow} \supInter{\mathfrak{ff}}\\
    \quad&=\quad \supInter{\mathfrak{ff}} \supInter{\uparrow} \supInter{\mathfrak{ff}}\\
    \quad&=\quad \supInter{\mathfrak{tt}}
  \end{flalign*}
  %
\end{frame}
%
\begin{frame}{Denotational semantics: statements}
  \begin{itemize}
  \item Now, we turn our heads to statements.
  \end{itemize}
  %
  \begin{block}{Denotational semantics of statements}
    $\pmb{\mathscr{S}} : \mathbb{V} \to (\texttt{S} \in \mathbb{S})
    \to \mathbb{Env} \rightharpoonup \mathbb{Env}
    %
    \text{\small\itshape~where } \mathbb{vars}(\texttt{S}) \subseteq
    \mathbb{V}$
    %
    \pause\begin{flalign*}
    \hspace*{-.5em}\begin{array}{llll}
      \ssem{\texttt{skip}} \sigma
      &\triangleq
      &\sigma
      &\\[.5em]
      %
      \ssem{\texttt{x := E}} \sigma
      &\triangleq
      &\sigma[v/\texttt{x}]
      &\text{%
        if $v = \esem{\texttt{E}}\sigma$
      }\\[.5em]
      %
      \ssem{\texttt{S}_1 \texttt{; } \texttt{S}_2}
      &\triangleq
      &\ssem{\texttt{S}_2} \circ \ssem{\texttt{S}_1}
      &\\[2em]
      %
      \ssem{\texttt{\textbf{if} (E) \textbf{then }} \texttt{S}_1 \texttt{\bfseries~else } \texttt{S}_2} \sigma
      &\triangleq
      &\begin{cases}
        \ssem{\texttt{S}_1} \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{tt}}\\
        \ssem{\texttt{S}_2} \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{ff}}\\
        undefined & otherwise
      \end{cases}
      &\\[.5em]
    \end{array}&&
    \end{flalign*}
  \end{block}
\end{frame}
%
\begin{frame}{Denotational semantics: loops}
  \begin{flalign*}
    \hspace*{-.5em}\begin{array}{lll}
      \ssem{\texttt{\textbf{while} (E) \textbf{do} S}} \sigma
      &\triangleq
      &\begin{cases}
        \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{ff}}\\
        (\ssem{\texttt{\only<2->{\color{red}}\textbf{while} (E) \textbf{do} S}} \circ \ssem{\texttt{S}}) \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{tt}}\\
        undefined & otherwise
      \end{cases}
    \end{array}&&
  \end{flalign*}
  %
  \begin{itemize}
  \pause\item \emph{\color{red}Recursive} definition!  \emph{E.g.}, $x = x + 1$.
  \pause\item We need to make sure that:
    \begin{enumerate}
    \item The equation above has (at least) one solution.
    \item If the equation has a solution, we must take the \emph{most precise} one.
    \end{enumerate}
  \pause\item And, for this, we use \emph{fixpoints} on \emph{posets}.
  \end{itemize}
  %
\end{frame}
%
\begin{frame}{Posets \& fixpoints: recap}
  \vspace*{-1em}
  \begin{itemize}
  \item We start by defining a poset $(\mathcal{I},\leq_{\mathcal{I}})$ with:
    %
    \begin{flalign*}
    \mathcal{I} &\triangleq \mathbb{I} \cup \{\bot\},\tag*{\scriptsize\color{gray} (values + undefined)}\\
    x \leq_{\mathcal{I}} y &\triangleq x = \bot \vee x = y.\tag*{\scriptsize\color{gray} (information order on values)}
    \end{flalign*}
    %
    That is, we add a bottom-most element $\bot$, the \emph{undefined}
    value and we define a \emph{flat order}, yielding a nicely
    structured poset: directed-complete, and finite chains.
  \end{itemize}
  %
  \begin{center}
  \begin{tikzpicture}[node distance=.3cm]
    \node (bot) {$\bot$};
    \node (dots) [above=of bot] {$\ldots$};
    \node (two) [left=of dots] {$2$};
    \node (one) [left=of two] {$1$};
    \node (tt) [right=of dots] {$\supInter{\mathfrak{tt}}$};
    \node (ff) [right=of tt] {$\supInter{\mathfrak{ff}}$};
    \draw (bot) -- (one);
    \draw (bot) -- (two);
    \draw (bot) -- (tt);
    \draw (bot) -- (ff);
  \end{tikzpicture}
  \end{center}
  %
  \begin{itemize}
    \pause\item Similarly, we define the poset $(\mathcal{T},\leq_{\mathcal{T}})$
    with:
    %
    \begin{flalign*}
    \mathcal{T} &\triangleq \mathbb{Env} \cup \{\bot\} \to \mathbb{Env} \cup \{\bot\},\tag*{\scriptsize\color{gray} (total, continuous transformers)}\\
      f \leq_{\mathcal{T}} g &\triangleq \forall \sigma\:\pmb{:}\: f(\sigma) \; \dot{\leq}_{\mathcal{I}} \; g(\sigma).\tag*{\scriptsize\color{gray} (information order on transformers)}
    \end{flalign*}
    %
    Where $\dot{\leq}_{\mathcal{I}}$ is the \emph{point-wise extension} of $\leq_{\mathcal{I}}$ to states and
    $\pmb{\lambda}\sigma\pmb{.}\; \bot$ is the bottom-most element.
    %
    This poset preserves all the nice properties of the one above.
  \end{itemize}
\end{frame}
%
\begin{frame}{Posets \& fixpoints: recap (cont'd)}
  \begin{itemize}
  \item A poset $(\mathcal{T},\leq_{\mathcal{T}})$ is a
    \emph{directed-complete partial order} (dcpo) if all of its
    directed subsets $D$ have a supremum $\vee_{\mathcal{T}}D$.
  \pause\item A function $f$ is \emph{Scott-continuous} if it preserves all directed
    suprema.
    %
    That is,
    $\forall D \subseteq \mathcal{T},~directed\;\pmb{:}\;f(\vee_{\mathcal{T}}D)
    = \vee_{\mathcal{T}}\{f(x)\;\pmb{:}\;x \in D\}.$
  \end{itemize}
  \pause\begin{theorem}[Kleene's theorem]
    Given a $\mathrm{directed}$-$\mathrm{complete~partial~order}$
    (dcpo) $(\mathcal{T},\leq_{\mathcal{T}})$, and a
    $\mathrm{Scott}$-$\mathrm{continuous~function}$
    $f : \mathcal{T} \to \mathcal{T}$, $f$ has a
    $\mathrm{least~fixpoint}$, which is the supremum:
    %
    \[
      \pmb{lfp}_{\leq_{\mathcal{T}}} f = \vee_{\mathcal{T}} \{f^n(\pmb{\lambda}\sigma\pmb{.}\;\bot)\;\pmb{:}\;n < \omega\}
    \]
    %
    of the ascending
    Kleene chain:
    %
    \[
      f^0(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}} f^1(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}} f^2(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}} \ldots \leq_{\mathcal{T}} f^n(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}} \ldots \tag*{$\triangleleft$}
    \]
  \end{theorem}
  %
  \vspace*{-1ex}
  %
  \pause So, our goal now is to make our denotational semantics function
  $\pmb{\mathscr{S}}$ to ``fit'' in this setting.
\end{frame}
%
\begin{frame}{Loop fixpoint semantics}
  \begin{enumerate}
  \item Extend the definitions of $\pmb{\mathscr{E}}$ and $\pmb{\mathscr{S}}$ with
    $\bot$ in the co-domain to model ``undefined'' behavior and turn them into \emph{total functions}:
    %
    \[
      \pmb{\mathscr{S}} : \mathbb{V} \to (\texttt{S} \in \mathbb{S})
      \to \mathbb{Env}\;{\color{MyPurple}\to}\;
      \mathbb{Env} \cup \{{\color{MyPurple}\bot}\}
      %
      \text{\small\it~where } \mathbb{vars}(\texttt{S}) \subseteq
      \mathbb{V}.
    \]
  \pause\item Make them composable (\emph{i.e.}, domain $=$ co-domain) by allowing
    $\bot$ in the domain:
    %
    \[
      \pmb{\mathscr{S}} : \mathbb{V} \to (\texttt{S} \in \mathbb{S})
      \to \mathbb{Env} \cup \{{\color{MyPurple}\bot}\} \to
      \mathbb{Env} \cup \{\bot\}
      %
      \text{\small\it~where } \mathbb{vars}(\texttt{S}) \subseteq
      \mathbb{V}.
    \]
    \pause\item Prove that
    $\ssem{\texttt{\_}} : \mathbb{Env} \cup \{\bot\} \to \mathbb{Env}
    \cup
    \{\bot\}$ is \emph{Scott-continuous}.\\
    {\small\color{gray}(Sketch: by structural induction on the
      program's syntax)}
  \end{enumerate}
\end{frame}
%
\begin{frame}{Loop fixpoint semantics (cont'd)}
  %
  Now, we can express the solution to the \texttt{\bfseries while}'s (semantic)
  equation as the \emph{least fixpoint} (\emph{i.e.}, most-precise for
  the information order) of the \emph{functional} $F$:
  %
  \begin{flalign*}
    \hspace*{-.5em}\begin{array}{lll}
      \ssem{\texttt{\textbf{while} (E) \textbf{do} S}} \sigma
      &\triangleq
      &\pmb{lfp}_{\leq_{\mathcal{T}}} F
    \end{array}&&
  \end{flalign*}
  %
  \vspace*{-2em}
  %
  \begin{flalign*}
    \hspace*{-.5em}\begin{array}{lll}
      \textit{where} &F : (\mathbb{Env} \cup \{\bot\} \to \mathbb{Env} \cup \{\bot\}) \to (\mathbb{Env} \cup \{\bot\} \to \mathbb{Env} \cup \{\bot\}),\qquad\qquad\text{\small\color{gray} (or, $F : \mathcal{T} \to \mathcal{T}$)}\\[1em]
      &F(f)(\sigma) = \begin{cases}
        \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{ff}}\\
        (f \circ \ssem{\texttt{S}}) \sigma & \text{if } \esem{\texttt{E}} \sigma = \supInter{\mathfrak{tt}}\\
        \bot & otherwise
      \end{cases}
    \end{array}&&
  \end{flalign*}
\end{frame}
%
\begin{frame}[fragile]{Loop fixpoint semantics: example}
  \begin{itemize}
  \item Consider computing the semantics of the program:
\begin{lstlisting}[basicstyle=\ttfamily,columns=fullflexible,escapeinside={(*}{*)}]
  (*\textbf{while}*) (0 < n) (*\textbf{do}*)
    x := x - n; n := n - 1
\end{lstlisting}
  \item We know that $\ssem{\texttt{x := x - n; n := n - 1}}\sigma = \sigma[x-n/x,n-1/n]$.
  \item We compute the least fixpoint of the functional $F$ defined as follows:
    %
    \[
      F(f)(\sigma) = \begin{cases}
        \sigma & \text{if } \esem{\texttt{0 < n}} \sigma = \supInter{\mathfrak{ff}}\\
        f \circ \sigma[x-n/x,n-1/n] & \text{if } \esem{\texttt{0 < n}} \sigma = \supInter{\mathfrak{tt}}\\
        \bot & otherwise
      \end{cases}
    \]
    %
    by building the chain $F^0(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}} F^1(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}} F^2(\pmb{\lambda}\sigma\pmb{.}\;\bot) \leq_{\mathcal{T}} \ldots \leq_{\mathcal{T}} \ldots$
  \end{itemize}
\end{frame}
%
\begin{frame}{Loop fixpoint semantics: example}
  \begin{flalign*}
    F^0(\pmb{\lambda}\sigma\pmb{.}\;\bot)(\sigma) = \bot
    && \text{\small\color{gray} (no information)}
  \end{flalign*}
\end{frame}

      \ssem{\texttt{S}} \sigma & \text{if } (\esem{\texttt{0 < n}} \circ \ssem{S}) \sigma = \supInter{\mathfrak{ff}}\\
      \sigma & \text{if } \esem{\texttt{0 < n}} \sigma = \supInter{\mathfrak{ff}}\\

    F^n(\pmb{\lambda}\sigma\pmb{.}\;\bot)(\sigma)
    && \text{\small\color{gray} (at most $n-1$ iterations)}
  \end{flalign*}
\end{frame}

\begin{frame}{Loop fixpoint semantics: example}
